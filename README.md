## Install

```bash
npm install --save react-float-tooltip
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-float-tooltip'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
